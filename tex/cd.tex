\def\CDdown#1#2#3{%
  \btexdraw
    \drawdim mm
    \linewd 0.1
    \move(0 0)
    \rlvec(6.5 0)
    \rlvec(0 118)
    \rlvec(-6.5 0)
    \rlvec(0 -118)

    \rmove(6.5 0)
    \rlvec(138 0)
    \rlvec(0 118)
    \rlvec(-138 0)
    \rmove(0 -118)

    \rmove(138 0)
    \rlvec(6.5 0)
    \rlvec(0 118)
    \rlvec(-6.5 0)

    \textref h:C v:C
    \vtext(3.25 59)
    {
      \hbox to 132mm
      {
        #1
      }
    }
    \htext(75.5 59)
    {
      \vbox to 112mm
      {
        \hsize=132mm
        #2
        \vfil
      }
    }
    \vtext(147.5 59)
    {
      \hbox to 132mm
      {
        #3
      }
    }
  \etexdraw
}
